## 1. 总体架构设计

### 1.1 架构概览

```
┌─────────────────────────────────────────────────────────┐
│                    PureAdmin 管理端                      │
├─────────────┬─────────────┬─────────────┬───────────────┤
│   视图层     │   状态层     │   路由层     │   请求层      │
│  (Vue3 SFC) │  (Pinia)    │  (vue-router)│  (axios)     │
├─────────────┴─────────────┴─────────────┴───────────────┤
│              PureAdmin 基础框架层                        │
│  (权限控制 / 布局 / 主题 / 国际化 / 指令 / 工具函数)      │
├─────────────────────────────────────────────────────────┤
│                    HTTP / Axios                         │
└─────────────────────────────────────────────────────────┘
                          │
                          ▼
┌─────────────────────────────────────────────────────────┐
│                   后端 API 网关                          │
│                  /api/v1/*                               │
└─────────────────────────────────────────────────────────┘
```

### 1.2 PureAdmin 框架集成

PureAdmin 是基于 Vue3 + TypeScript + Vite + Element Plus 的后台管理框架，提供：

| 能力         | 说明                                         |
| ------------ | -------------------------------------------- |
| **权限控制** | RBAC 权限模型，支持菜单权限、按钮权限、路由守卫 |
| **布局系统** | 多种布局模式（左侧菜单、顶部菜单、混合菜单）   |
| **主题系统** | 亮色/暗色主题，自定义主题色                   |
| **国际化**   | 中英文切换                                     |
| **请求封装** | 基于 Axios 的请求拦截、响应拦截、错误统一处理   |
| **组件库**   | Element Plus 二次封装，表格、表单、弹窗等常用组件 |
| **状态管理** | Pinia 状态管理，支持持久化                     |

// src/store/modules/auth.ts
import { defineStore } from "pinia";
import { getAccessToken, setAccessToken, removeAccessToken } from "@/utils/auth";
import { login, getUserInfo, logout } from "@/api/auth";

interface AuthState {
  token: string;
  userInfo: AdminUserInfo | null;
  roles: string[];
}

export const useAuthStore = defineStore("auth", {
  state: (): AuthState => ({
    token: getAccessToken(),
    userInfo: null,
    roles: []
  }),

  getters: {
    isLoggedIn: (state) => !!state.token,
    isAdmin: (state) => state.userInfo?.role === "super_admin",
    stationId: (state) => state.userInfo?.station_id
  },

  actions: {
    async login(credentials: LoginRequest) {
      const res = await login(credentials);
      this.token = res.access_token;
      setAccessToken(res.access_token);
      this.userInfo = res.user;
      return res;
    },

    async fetchUserInfo() {
      const res = await getUserInfo();
      this.userInfo = res;
      this.roles = [res.role];
      return res;
    },

    async logout() {
      try {
        await logout();
      } finally {
        this.resetState();
        removeAccessToken();
      }
    },

    resetState() {
      this.token = "";
      this.userInfo = null;
      this.roles = [];
    }
  }
});

### 8.1 Axios 配置

```typescript
// src/api/index.ts
import axios from "axios";
import { ElMessage } from "element-plus";
import { getAccessToken, removeAccessToken } from "@/utils/auth";
import router from "@/router";

const service = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 15000,
  headers: {
    "Content-Type": "application/json; charset=utf-8"
  }
});

// 请求拦截器
service.interceptors.request.use(
  (config) => {
    const token = getAccessToken();
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    config.headers["X-Client-Type"] = "admin";
    config.headers["X-Client-Version"] = "1.0.0";
    return config;
  },
  (error) => Promise.reject(error)
);

// 响应拦截器
service.interceptors.response.use(
  (response) => {
    const res = response.data;
    if (res.code !== 0) {
      ElMessage.error(res.msg || "请求失败");
      // Token 失效
      if (res.code === 10002) {
        removeAccessToken();
        router.push("/login");
      }
      return Promise.reject(new Error(res.msg));
    }
    return res.data;
  },
  (error) => {
    if (error.response?.status === 401) {
      removeAccessToken();
      router.push("/login");
    }
    ElMessage.error(error.message || "网络异常");
    return Promise.reject(error);
  }
);

export default service;
```

// src/hooks/usePermission.ts
import { useAuthStore } from "@/store/modules/auth";

export function usePermission() {
  const authStore = useAuthStore();

  /** 是否为系统管理员 */
  const isSuperAdmin = computed(() => authStore.userInfo?.role === "super_admin");

  /** 是否为驿站管理员 */
  const isStationAdmin = computed(() => authStore.userInfo?.role === "station_admin");

  /** 当前用户所属驿站ID */
  const stationId = computed(() => authStore.userInfo?.station_id);

  /** 检查是否有指定权限 */
  function hasPermission(permission: string): boolean {
    const roles = authStore.roles;
    // 系统管理员拥有所有权限
    if (roles.includes("super_admin")) return true;
    // 根据角色-权限映射判断
    return rolePermissionMap[roles[0]]?.includes(permission) ?? false;
  }

  /** 检查是否可操作指定驿站的数据 */
  function canAccessStation(targetStationId: number): boolean {
    if (isSuperAdmin.value) return true;
    return stationId.value === targetStationId;
  }

  return {
    isSuperAdmin,
    isStationAdmin,
    stationId,
    hasPermission,
    canAccessStation
  };
}

import { ref, reactive } from "vue";
import type { Ref } from "vue";

interface UseTableOptions<T> {
  fetchApi: (params: any) => Promise<{ list: T[]; total: number }>;
  immediate?: boolean;
}

export function useTable<T>(options: UseTableOptions<T>) {
  const { fetchApi, immediate = true } = options;

  const loading = ref(false);
  const tableData: Ref<T[]> = ref([]);
  const total = ref(0);
  const queryParams = reactive({
    page: 1,
    page_size: 20,
    ...extraParams
  });

  const fetchData = async () => {
    loading.value = true;
    try {
      const res = await fetchApi(queryParams);
      tableData.value = res.list;
      total.value = res.total;
    } finally {
      loading.value = false;
    }
  };

  const handlePageChange = (page: number) => {
    queryParams.page = page;
    fetchData();
  };

  const handleSizeChange = (size: number) => {
    queryParams.page_size = size;
    queryParams.page = 1;
    fetchData();
  };

  const handleSearch = () => {
    queryParams.page = 1;
    fetchData();
  };

  const handleReset = () => {
    Object.assign(queryParams, { page: 1, page_size: 20, ...defaultParams });
    fetchData();
  };

  if (immediate) fetchData();

  return {
    loading,
    tableData,
    total,
    queryParams,
    fetchData,
    handlePageChange,
    handleSizeChange,
    handleSearch,
    handleReset
  };
}
